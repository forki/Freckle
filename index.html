let sampler count =
    sample {
        let! currentSpan = Sample.period
        printfn "%d: Sampling %A -> %A" count (Time.toDateTime currentSpan.Beginning) (Time.toDateTime currentSpan.Finish)
        return count + 1
    } |> SampleAsync.ofSample

Sample.sampleForever Clock.systemUtc sampler 0
|> Async.RunSynchronously

let render =
    async {

let renderingSampler () =
    sample {
        let! pulses = Feed.pulseUpto 1u //Notice it's pulseUpto
        return! Feed.transition (fun _ _ -> render) () pulses
    }

|> Async.RunSynchronously

//Use an event queue of your choice(not included)
let eventQueue = BufferBlock() //Here I will use BufferBlock but use whatever you want

let dequeue =
    async {
        let! msg = eventQueue.ReceiveAsync() |> Async.AwaitTask
        return msg
    }

async {
    let! mb = Mailbox.createWithExpiration (Expire.After (Time.ofSeconds 30))  Clock.systemUtc
    do! Mailbox.listenTo dequeue mb

    let renderingSampler mb () =
        sampleAsync {
            let! events = Mailbox.read mb |> SampleAsync.ofAsync
            return! Feed.foldPast (fun _ msg -> printfn "revieved %s" msg) () events
                    |> SampleAsync.ofSample
        }

    do! Sample.sampleForever Clock.systemUtc (renderingSampler mb) ()
}
|> Async.RunSynchronously
